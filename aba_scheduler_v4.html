function getClient(id){

function timeToMinutes(time){


  if(!time) return 0;


  const parts =

    time.split(":");


  return (

    Number(parts[0]) * 60 +

    Number(parts[1])

  );

}


function durationHours(session){


  return (

    timeToMinutes(session.end_time) -

    timeToMinutes(session.start_time)

  ) / 60;

}


function clientHours(clientId){


  return sessions

    .filter(

      s =>

        s.client_id === clientId &&

        s.status === "scheduled"

    )

    .reduce(

      (sum,s) =>

        sum + durationHours(s),

      0

    );

}


function staffHours(staffId){


  return sessions

    .filter(

      s =>

        s.staff_id === staffId &&

        s.status === "scheduled"

    )

    .reduce(

      (sum,s) =>

        sum + durationHours(s),

      0

    );

}


function totalHours(){


  return sessions

    .filter(

      s =>

        s.status === "scheduled"

    )

    .reduce(

      (sum,s) =>

        sum + durationHours(s),

      0

    );

}


/* =========================

   OVERLAP CHECKS

========================= */


function overlaps(a,b){


  if(

    a.id === b.id ||

    a.day !== b.day ||

    a.staff_id !== b.staff_id

  ){

    return false;

  }


  return (

    timeToMinutes(a.start_time) <

    timeToMinutes(b.end_time)

    &&

    timeToMinutes(b.start_time) <

    timeToMinutes(a.end_time)

  );

}


function hasOverlap(session){


  return sessions.some(

    other =>

      overlaps(session,other)

  );

}

async function saveSession(){

  const id = sessionId.value;


  const payload = {

    client_id: sessionClient.value,

    staff_id: sessionStaff.value || null,

    day: sessionDay.value,

    slot: sessionSlot.value,

    start_time: sessionStart.value,

    end_time: sessionEnd.value,

    service_type: sessionService.value,

    status: sessionStatus.value,

    updated_at: new Date().toISOString()

  };


  if(!payload.staff_id){

    payload.status = "unscheduled";

  }


  if(id){

    await supabaseClient

      .from("sessions")

      .update(payload)

      .eq("id", id);

  } else {

    await supabaseClient

      .from("sessions")

      .insert(payload);

  }


  closeModal();

  await loadData();

}


async function deleteSession(){

  if(!sessionId.value) return closeModal();


  if(!confirm("Delete this appointment?")) return;


  await supabaseClient

    .from("sessions")

    .delete()

    .eq("id", sessionId.value);


  closeModal();

  await loadData();

}


/* DUPLICATE SESSION */


async function duplicateSession(){

  if(!sessionId.value){

    alert("Open an existing appointment first.");

    return;

  }


  const original =

    sessions.find(x => x.id === sessionId.value);


  if(!original) return;


  const input = prompt(

    "Copy to which day(s)? Example: W,F or Wednesday,Friday"

  );


  if(!input) return;


  const map = {

    m:"Monday", mon:"Monday", monday:"Monday",

    t:"Tuesday", tue:"Tuesday", tuesday:"Tuesday",

    w:"Wednesday", wed:"Wednesday", wednesday:"Wednesday",

    th:"Thursday", thu:"Thursday", thursday:"Thursday",

    f:"Friday", fri:"Friday", friday:"Friday"

  };


  const rows = input

    .split(",")

    .map(x => x.trim().toLowerCase())

    .map(x => map[x])

    .filter(Boolean)

    .map(day => ({

      client_id: original.client_id,

      staff_id: original.staff_id,

      day,

      slot: original.slot,

      start_time: original.start_time,

      end_time: original.end_time,

      service_type: original.service_type,

      status: original.status,

      notes: original.notes || ""

    }));


  if(rows.length){

    await supabaseClient

      .from("sessions")

      .insert(rows);

  }


  closeModal();

  await loadData();

}


/* RENAME STAFF */


async function renameStaff(staffId){

  const s = staff.find(x => x.id === staffId);

  if(!s) return;


  const newName = prompt("Edit staff name:", s.name);


  if(!newName || !newName.trim()) return;


  await supabaseClient

    .from("staff")

    .update({ name: newName.trim() })

    .eq("id", staffId);


  await loadData();

}
